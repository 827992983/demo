struct SegMent {
    WORD selector; //来自段选择子的索引部分
    WORD attribute; //attribute 属性记录了该段是否有效，是否可读可写可执行等权限
    DWORD base; //基地址，如：ds:[0x00012345]有效地址就是ds.base+0x123456得出具体地址
    DWORD limit; //标识这个段有多大,越界访问会报错
}
